\subsubsection\
$
\varphi =   (\Diamond K \rightarrow \Box B) \land
            (\Diamond \neg K \rightarrow \Box K) \land
            (\Diamond \Diamond (K \rightarrow B))\\
$
\subsubsection\
NNF bilden:\\
$
\varphi =   (\Diamond K \rightarrow \Box B) \land
            (\Diamond \neg K \rightarrow \Box K) \land
            (\Diamond \Diamond (K \rightarrow B))\\
\equiv      (\neg \Diamond K \lor \Box B) \land
            (\neg \Diamond \neg K \lor \Box K) \land
            (\Diamond \Diamond (\neg K \lor B))\\
\equiv      (\Box (\neg K \lor B)) \land 
            (\Box (K \lor B)) \land
            (\Diamond \Diamond (\neg K \lor B))
= \varphi'\\
$
\\

$
\begin{matrix}
        &&&s1, \varphi'\\
