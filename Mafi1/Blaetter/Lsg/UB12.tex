f(v+w)  = f(1,1,2)^t\\
        = (3,3,2)^t \neq (3,3,1)^t\\
        = (3,3,1)^t + (0,0,0)^t\\
        = f(v) + f(w)\\
$
\\
b)\\
$
f(v+w)  = (-l(v+w), (l(v+w) + l(2*(v+w))))\\
        = ((-l(v) + -l(w)), (l(2v) + l(2w)))\\
        = (-l(v), l(2v)) + (-l(w), l(2w))\\
        = f(v) + f(w)\\
\\
f(s*v)  = (-l(s*v), (l(s*v) + l(s*2*v)))\\
        = (s*-l(s*v), (s*l(v) + s*l(2*v)))\\
        = s * f(v)\\
$
